function erreurPosition(erreur) { switch(erreur.code) { case erreur.PERMISSION_DENIED: alert("PERMISSION_DENIED : l'utilisateur n'a pas autorisé l'accès à sa position."); break; case erreur.POSITION_UNAVAILABLE: alert("POSITION_UNAVAILABLE : la position n'a pas pu être déterminée."); break; case erreur.TIMEOUT: alert("TIMEOUT : le service n'a pas répondu à temps."); break; case erreur.UNKNOWN_ERROR: alert("UNKNOWN_ERROR : une erreur inconnue s'est produite");} }
